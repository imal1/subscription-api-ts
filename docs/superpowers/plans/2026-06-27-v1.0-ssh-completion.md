- 单机模式（无 nodes.yaml）行为不变

- GitHub Releases 作为 Agent 二进制分发源
- Agent 状态机: not_deployed → deploying → running/error → stopped → updating → uninstalling

```typescript
export class DeployManager {
  private static instance: DeployManager;
  private sshPool: Map<string, NodeSSH> = new Map();
  private connectingPromises: Map<string, Promise<NodeSSH>> = new Map();

  // ... existing methods ...

  /** 获取 SSH 私钥存储路径 */
  getSshKeyPath(nodeId: string): string {
    return path.join(os.homedir(), '.config', 'miobridge', 'ssh', `${nodeId}.key`);
  }

  /** 建立 SSH 连接（带连接池和去重） */
  async connect(nodeId: string): Promise<NodeSSH> {
    // 复用已有连接
    const existing = this.sshPool.get(nodeId);
    if (existing && existing.isConnected()) {
      return existing;
    }

    // 去重：同一 nodeId 的并发 connect 共享一个 Promise
    const pending = this.connectingPromises.get(nodeId);
    if (pending) return pending;

    const connectPromise = this.doConnect(nodeId);
    this.connectingPromises.set(nodeId, connectPromise);

    try {
      const ssh = await connectPromise;
      this.sshPool.set(nodeId, ssh);
      return ssh;
    } finally {
      this.connectingPromises.delete(nodeId);
    }
  }

  private async doConnect(nodeId: string): Promise<NodeSSH> {
    const nodeManager = NodeManager.getInstance();
    const nodes = await nodeManager.loadNodes();
    const node = nodes.find(n => n.id === nodeId);

    if (!node || !node.ssh) {
      throw new Error(`节点 ${nodeId} 未配置 SSH 信息`);
    }

    const keyPath = node.ssh.keyPath || this.getSshKeyPath(nodeId);
    if (!fs.existsSync(keyPath)) {
      throw new Error(`SSH 私钥不存在: ${keyPath}`);
    }

    const ssh = new NodeSSH();
    await ssh.connect({
      host: node.host,
      port: node.ssh.port || 22,
      username: node.ssh.user || 'root',
      privateKeyPath: keyPath,
      readyTimeout: 15000,
      timeout: 10000,
      // 首次连接自动接受 host key
      ...(node.ssh.hostKey ? { hostVerifier: (hashedKey: string) => hashedKey === node.ssh.hostKey } : {}),
    });

    return ssh;
  }

  /** 断开 SSH 连接 */
  disconnect(nodeId: string): void {
    const ssh = this.sshPool.get(nodeId);
    if (ssh) {
      ssh.dispose();
      this.sshPool.delete(nodeId);
    }
  }

  /** 断开所有连接 */
  disconnectAll(): void {
    for (const [nodeId] of this.sshPool) {
      this.disconnect(nodeId);
    }
  }

  /** 解析 OS 信息（uname -a + cat /etc/os-release 输出） */
  parseOsInfo(raw: string): { os: string; arch: string } {
    const lines = raw.trim().split('\n').filter(Boolean);
    let os = 'unknown';
    let arch = 'unknown';

    for (const line of lines) {
      const trimmed = line.trim().toLowerCase();
      if (trimmed.includes('x86_64') || trimmed.includes('amd64')) arch = 'x86_64';
      else if (trimmed.includes('aarch64') || trimmed.includes('arm64')) arch = 'aarch64';
      else if (trimmed.includes('armv7')) arch = 'armv7';

      if (trimmed.includes('ubuntu')) os = line.trim();
      else if (trimmed.includes('debian')) os = line.trim();
      else if (trimmed.includes('centos')) os = line.trim();
    }

    return { os, arch };
  }

  /** 测试 SSH 连接并检测远程系统信息 */
  async testConnection(nodeId: string): Promise<{ ok: boolean; os: string; arch: string; bunVersion?: string; error?: string }> {
    let ssh: NodeSSH | null = null;
    try {
      ssh = await this.connect(nodeId);

      // 获取 OS 信息
      const osResult = await ssh.execCommand(
        'cat /etc/os-release 2>/dev/null | head -5 && uname -m'
      );
      const { os, arch } = this.parseOsInfo(osResult.stdout);

      // 检测 Bun 是否已安装
      let bunVersion: string | undefined;
      const bunResult = await ssh.execCommand('bun --version 2>/dev/null || echo "NOT_FOUND"');
      if (bunResult.stdout && !bunResult.stdout.includes('NOT_FOUND')) {
        bunVersion = bunResult.stdout.trim();
      }

      return { ok: true, os, arch, bunVersion };
    } catch (error: any) {
      return { ok: false, os: 'unknown', arch: 'unknown', error: error.message };
    } finally {
      // testConnection 后不保持连接（按需连接）
      if (ssh) this.disconnect(nodeId);
    }
  }
}
```

/** Bun 安装命令 */
getBunInstallCmd(): string {
  return 'curl -fsSL https://bun.sh/install | bash';
}

/** 获取部署步骤顺序 */
getDeploySteps(): string[] {
  return ['connect', 'bun', 'kernel', 'agent', 'start', 'verify'];
}

/** GitHub Releases Agent 二进制下载 URL */
getAgentDownloadUrl(version: string, arch: string): string {
  const archSuffix = arch === 'aarch64' || arch === 'arm64' ? 'arm64' : 'amd64';
  return `https://github.com/imal1/miobridge/releases/download/${version}/miobridge-agent-linux-${archSuffix}`;
}

/** 完整部署流程（AsyncGenerator，供 SSE 消费） */
async *fullDeploy(nodeId: string): AsyncGenerator<DeployStep> {
  const nodeManager = NodeManager.getInstance();
  const nodes = await nodeManager.loadNodes();
  const node = nodes.find(n => n.id === nodeId);
  if (!node) throw new Error(`节点 ${nodeId} 不存在`);

  // 更新状态为 deploying
  await nodeManager.updateNodeStatus(nodeId, 'deploying', '用户触发部署');

  let ssh: NodeSSH | null = null;

  try {
    // Step 1: 连接 + 系统检测
    yield { step: 'connect', status: 'running', message: '正在 SSH 连接...', progress: 0 };
    const connResult = await this.testConnection(nodeId);
    if (!connResult.ok) {
      yield { step: 'connect', status: 'error', message: `连接失败: ${connResult.error}`, progress: 0 };
      await nodeManager.updateNodeStatus(nodeId, 'error', `SSH 连接失败: ${connResult.error}`);
      return;
    }
    yield { step: 'connect', status: 'success', message: `已连接 (${connResult.os}, ${connResult.arch})`, progress: 10 };

    ssh = await this.connect(nodeId);

    // Step 2: 安装 Bun
    yield { step: 'bun', status: 'running', message: '正在安装 Bun...', progress: 15 };
    if (!connResult.bunVersion) {
      const bunResult = await ssh.execCommand(this.getBunInstallCmd(), { timeout: 120000 });
      if (bunResult.code !== 0) {
        yield { step: 'bun', status: 'error', message: `Bun 安装失败: ${bunResult.stderr}`, progress: 25 };
        await nodeManager.updateNodeStatus(nodeId, 'error', `Bun 安装失败`);
        return;
      }
      // source profile 使 bun 可用
      await ssh.execCommand('export PATH="$HOME/.bun/bin:$PATH" && hash -r');
    }
    yield { step: 'bun', status: 'success', message: `Bun ${connResult.bunVersion || '已安装'}`, progress: 25 };

    // Step 3: 安装内核
    yield { step: 'kernel', status: 'running', message: `正在安装 ${node.kernel}...`, progress: 30 };
    const kernelCmd = this.getKernelInstallCmd(node.kernel);
    const kernelResult = await ssh.execCommand(kernelCmd, { timeout: 300000 });
    if (kernelResult.code !== 0) {
      yield { step: 'kernel', status: 'error', message: `内核安装失败: ${kernelResult.stderr}`, progress: 50 };
      await nodeManager.updateNodeStatus(nodeId, 'error', `内核安装失败`);
      return;
    }
    yield { step: 'kernel', status: 'success', message: `${node.kernel} 安装完成`, progress: 50 };

    // Step 4: 部署 Agent
    yield { step: 'agent', status: 'running', message: '正在部署 Agent...', progress: 55 };

    const secret = this.generateHmacSecret();
    const agentYaml = this.generateAgentYaml(node.id, node.name, secret, node.kernel);
    const systemdUnit = this.generateSystemdUnit(secret);
    const arch = connResult.arch;

    // 下载 Agent 二进制（从 GitHub Releases 或控制面）
    const version = await this.getLatestReleaseVersion();
    const downloadUrl = this.getAgentDownloadUrl(version, arch);

    // 确保目标目录存在
    await ssh.execCommand('mkdir -p /etc/miobridge-agent /usr/local/bin');

    // 下载 Agent 二进制
    const downloadResult = await ssh.execCommand(
      `curl -fsSL '${downloadUrl}' -o /usr/local/bin/miobridge-agent && chmod +x /usr/local/bin/miobridge-agent`,
      { timeout: 120000 }
    );
    if (downloadResult.code !== 0) {
      yield { step: 'agent', status: 'error', message: `Agent 下载失败: ${downloadResult.stderr}`, progress: 75 };
      await nodeManager.updateNodeStatus(nodeId, 'error', `Agent 下载失败`);
      return;
    }

    // 上传 agent.yaml
    await ssh.execCommand(`cat > /etc/miobridge-agent/agent.yaml << 'AGENTEOF'\n${agentYaml}\nAGENTEOF`);
    await ssh.execCommand('chmod 600 /etc/miobridge-agent/agent.yaml');

    // 上传 systemd unit
    await ssh.execCommand(`cat > /etc/systemd/system/miobridge-agent.service << 'UNITEOF'\n${systemdUnit}\nUNITEOF`);

    yield { step: 'agent', status: 'success', message: 'Agent 部署完成', progress: 75 };

    // Step 5: 启动服务
    yield { step: 'start', status: 'running', message: '正在启动 Agent...', progress: 80 };
    await ssh.execCommand('systemctl daemon-reload && systemctl enable --now miobridge-agent');
    // 等待服务启动
    await new Promise(resolve => setTimeout(resolve, 3000));
    yield { step: 'start', status: 'success', message: 'Agent 已启动', progress: 90 };

    // Step 6: 健康验证
    yield { step: 'verify', status: 'running', message: '正在进行健康检查...', progress: 92 };
    const healthResult = await ssh.execCommand('curl -fsS --max-time 5 http://127.0.0.1:3001/health 2>/dev/null');
    if (healthResult.code !== 0) {
      yield { step: 'verify', status: 'error', message: `健康检查失败: ${healthResult.stderr}`, progress: 100 };
      await nodeManager.updateNodeStatus(nodeId, 'error', 'Agent 启动后健康检查失败');
      return;
    }
    yield { step: 'verify', status: 'success', message: '健康检查通过', progress: 100 };

    // 更新 nodes.yaml 中 agent 状态
    await nodeManager.updateNodeStatus(nodeId, 'running', '部署成功，健康检查通过');
    await nodeManager.updateNodeAgent(nodeId, {
      deployed: true,
      version,
      status: 'running',
      lastDeploy: new Date().toISOString(),
    });

    yield { step: 'done', status: 'success', message: '部署完成', progress: 100 };

  } catch (error: any) {
    yield { step: 'done', status: 'error', message: `部署异常: ${error.message}`, progress: 0 };
    await nodeManager.updateNodeStatus(nodeId, 'error', error.message);
  } finally {
    if (ssh) this.disconnect(nodeId);
  }
}
```

/** 查询 GitHub Releases 最新版本 */
async getLatestReleaseVersion(): Promise<string> {
  try {
    const response = await fetch(this.GITHUB_RELEASES_API, {
      headers: {
        'Accept': 'application/vnd.github+json',
        'User-Agent': 'miobridge-agent',
      },
    });
    if (!response.ok) {
      // Fallback: 使用当前 package.json 版本
      return 'v1.0.0';
    }
    const data = await response.json() as any;
    return data.tag_name || 'v1.0.0';
  } catch {
    return 'v1.0.0';
  }
}

/** 检查哪些节点需要更新 Agent */
async checkAgentUpdates(): Promise<Array<{ nodeId: string; current: string; latest: string }>> {
  const latest = await this.getLatestReleaseVersion();
  const nodeManager = NodeManager.getInstance();
  const nodes = await nodeManager.loadNodes();

  return nodes
    .filter(n => n.agent?.deployed && n.agent.version && n.agent.version !== latest)
    .map(n => ({ nodeId: n.id, current: n.agent!.version, latest }));

/** 更新 Agent（从 GitHub Releases 下载新版本，替换后重启） */
async updateAgent(nodeId: string): Promise<{ ok: boolean; message: string }> {
  let ssh: NodeSSH | null = null;
  try {
    const nodeManager = NodeManager.getInstance();
    await nodeManager.updateNodeStatus(nodeId, 'updating', '用户触发 Agent 更新');

    ssh = await this.connect(nodeId);
    const connResult = await this.testConnection(nodeId);
    const arch = connResult.arch;
    const version = await this.getLatestReleaseVersion();
    const downloadUrl = this.getAgentDownloadUrl(version, arch);

    // 下载新版本
    const result = await ssh.execCommand(
      `curl -fsSL '${downloadUrl}' -o /usr/local/bin/miobridge-agent.new && chmod +x /usr/local/bin/miobridge-agent.new`,
      { timeout: 120000 }
    );
    if (result.code !== 0) {
      await nodeManager.updateNodeStatus(nodeId, 'error', `Agent 更新下载失败: ${result.stderr}`);
      return { ok: false, message: `下载失败: ${result.stderr}` };
    }

    // 替换 + 重启
    await ssh.execCommand('systemctl stop miobridge-agent');
    await ssh.execCommand('mv /usr/local/bin/miobridge-agent.new /usr/local/bin/miobridge-agent');
    await ssh.execCommand('systemctl start miobridge-agent');

    // 健康验证
    await new Promise(resolve => setTimeout(resolve, 3000));
    const healthResult = await ssh.execCommand('curl -fsS --max-time 5 http://127.0.0.1:3001/health 2>/dev/null');

    if (healthResult.code === 0) {
      await nodeManager.updateNodeStatus(nodeId, 'running', `更新到 ${version} 成功`);
      await nodeManager.updateNodeAgent(nodeId, { version, lastDeploy: new Date().toISOString() });
      return { ok: true, message: `已更新到 ${version}` };
    } else {
      await nodeManager.updateNodeStatus(nodeId, 'error', `更新后健康检查失败`);
      return { ok: false, message: '更新后健康检查失败' };
    }
  } catch (error: any) {
    await NodeManager.getInstance().updateNodeStatus(nodeId, 'error', error.message);
    return { ok: false, message: error.message };
  } finally {
    if (ssh) this.disconnect(nodeId);
  }
}

  'update': 'updating',
  'restart': 'running',
  'uninstall': 'uninstalling',
  'redeploy': 'deploying',
};

/** 检查状态转换是否合法 */
static canTransition(currentStatus: string, action: string): boolean {
  const allowed = NodeManager.TRANSITION_RULES[action];
  if (!allowed) return false;
  return allowed.includes(currentStatus);
}

/** 更新节点 Agent 状态（带状态守卫和 statusHistory） */
async updateNodeStatus(nodeId: string, newStatus: string, reason: string): Promise<void> {
  await this.loadNodes();
  const node = this.nodes.find(n => n.id === nodeId);
  if (!node) return;

  const currentStatus = node.agent?.status || 'not_deployed';
  const history: AgentStatusTransition[] = node.agent?.statusHistory || [];

  history.push({
    from: currentStatus,
    to: newStatus,
    at: new Date().toISOString(),
    reason,
  });

  // 保留最近 10 条
  if (history.length > 10) history.shift();

  if (!node.agent) {
    node.agent = { deployed: false, version: '', status: newStatus as any, lastDeploy: '' };
  } else {
    node.agent.status = newStatus as any;
  }
  node.agent.statusHistory = history;
  if (newStatus === 'error') node.agent.lastError = reason;

  await this.writeNodesYaml();
}

/** 更新节点 Agent 字段 */
async updateNodeAgent(nodeId: string, patch: Partial<NodeAgentInfo>): Promise<void> {
  await this.loadNodes();
  const node = this.nodes.find(n => n.id === nodeId);
  if (!node) return;

  if (!node.agent) {
    node.agent = { deployed: false, version: '', status: 'not_deployed', lastDeploy: '' };
  }

  Object.assign(node.agent, patch);
  await this.writeNodesYaml();
}

/** 序列化并写入 nodes.yaml */
async writeNodesYaml(): Promise<void> {
  const dir = path.dirname(NODES_YAML_PATH);
  if (!fs.existsSync(dir)) {
    fs.mkdirpSync(dir);
  }

  const yamlContent = this.serializeNodesYaml();
  await fs.writeFile(NODES_YAML_PATH, yamlContent, 'utf8');
  logger.info(`NodeManager: nodes.yaml 已更新 (${this.nodes.length} 个节点)`);
}

/** 序列化 nodes 为 YAML */
private serializeNodesYaml(): string {
  const lines: string[] = ['nodes:'];

  for (const node of this.nodes) {
    lines.push(`  - id: "${node.id}"`);
    if (node.name) lines.push(`    name: "${node.name}"`);
    if (node.host) lines.push(`    host: "${node.host}"`);
    lines.push(`    port: ${node.port || 443}`);
    if (node.secret) lines.push(`    secret: "${node.secret}"`);
    if (node.kernel) lines.push(`    kernel: "${node.kernel}"`);
    if (node.location) lines.push(`    location: "${node.location}"`);
    lines.push(`    enabled: ${node.enabled !== false}`);

    if (node.ssh) {
      lines.push('    ssh:');
      lines.push(`      user: "${node.ssh.user}"`);
      lines.push(`      port: ${node.ssh.port || 22}`);
      lines.push(`      keyPath: "${node.ssh.keyPath}"`);
      lines.push(`      hostKey: "${node.ssh.hostKey}"`);
    }

    if (node.agent) {
      lines.push('    agent:');
      lines.push(`      deployed: ${node.agent.deployed}`);
      lines.push(`      version: "${node.agent.version}"`);
      lines.push(`      status: "${node.agent.status}"`);
      lines.push(`      lastDeploy: "${node.agent.lastDeploy}"`);
      if (node.agent.lastError) lines.push(`      lastError: "${node.agent.lastError}"`);
      if (node.agent.statusHistory && node.agent.statusHistory.length > 0) {
        lines.push('      statusHistory:');
        for (const h of node.agent.statusHistory) {
          lines.push(`        - from: "${h.from}"`);
          lines.push(`          to: "${h.to}"`);
          lines.push(`          at: "${h.at}"`);
          lines.push(`          reason: "${h.reason}"`);
        }
      }
    }

    if (node.kernelInfo) {
      lines.push('    kernelInfo:');
      lines.push(`      installed: ${node.kernelInfo.installed}`);
      lines.push(`      version: "${node.kernelInfo.version}"`);
      lines.push(`      installScript: "${node.kernelInfo.installScript}"`);
    }
  }

  return lines.join('\n') + '\n';
}

/** 添加节点到 nodes.yaml */
async addNode(config: NodeConfig): Promise<void> {
  await this.loadNodes();
  const existing = this.nodes.findIndex(n => n.id === config.id);
  if (existing >= 0) {
    this.nodes[existing] = { ...this.nodes[existing], ...config };
  } else {
    this.nodes.push(config);
  }
  await this.writeNodesYaml();
  logger.info(`NodeManager: 添加/更新节点 ${config.id}`);
}

/** 从 nodes.yaml 删除节点 */
async removeNode(nodeId: string): Promise<void> {
  await this.loadNodes();
  this.nodes = this.nodes.filter(n => n.id !== nodeId);
  this.nodeCache.delete(nodeId);
  await this.writeNodesYaml();
  logger.info(`NodeManager: 删除节点 ${nodeId}`);
}

import type { NextApiRequest, NextApiResponse } from 'next';
import { DeployManager } from '@/server/services/deployManager';
import { logger } from '@/server/utils/logger';

export default async function handler(req: NextApiRequest, res: NextApiResponse) {
  const nodeId = (req.query.node as string) || '';

  if (!nodeId) {
    res.status(400).json({ error: '缺少 node 参数' });
    return;
  }

  // SSE headers
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache, no-transform');
  res.setHeader('Connection', 'keep-alive');
  res.setHeader('X-Accel-Buffering', 'no'); // nginx 禁用缓冲
  res.flushHeaders();

  let closed = false;

  req.on('close', () => {
    closed = true;
  });

  try {
    const deployManager = DeployManager.getInstance();

    for await (const step of deployManager.fullDeploy(nodeId)) {
      if (closed) break;

      const data = JSON.stringify(step);
      res.write(`data: ${data}\n\n`);

      // 部署完成或出错时发送 done 事件
      if (step.step === 'done' || step.status === 'error') {
        res.write(`event: done\ndata: ${data}\n\n`);
        break;
      }
    }
  } catch (error: any) {
    logger.error('SSE 部署进度异常:', error);
    if (!closed) {
      res.write(`data: ${JSON.stringify({ step: 'done', status: 'error', message: error.message, progress: 0 })}\n\n`);
    }
  } finally {
    if (!closed) {
      res.end();
    }
  }
}

### Agent 部署失败

**症状**：Dashboard 点击"一键部署"后进度卡在某一步。

**排查**：
```bash
# 检查 SSH 连接
ssh -i ~/.config/miobridge/ssh/<node-id>.key root@<host>

# 检查 Agent 服务状态
systemctl status miobridge-agent

# 查看 Agent 日志
journalctl -u miobridge-agent -n 50
```

**常见原因**：
- SSH 端口不通 → 检查防火墙/安全组
- 私钥权限不对 → `chmod 600 ~/.config/miobridge/ssh/<node-id>.key`
- GitHub Releases 下载慢 → 检查远程节点 GitHub 连通性
- Bun 安装失败 → 手动 `curl -fsSL https://bun.sh/install | bash`

### 节点状态异常

**症状**：节点卡片显示"异常"状态。

**排查**：
```bash
# 查看状态历史
cat ~/.config/miobridge/nodes.yaml | grep -A 20 "id: <node-id>"

# 检查 Agent 是否运行
curl https://<host>/api/health
```

### SSH 连接失败

**症状**：testConnection 返回失败。

**排查**：
- 确认 SSH 私钥路径正确：`~/.config/miobridge/ssh/<node-id>.key`
- 确认远程节点 SSH 服务运行：`systemctl status sshd`
- 确认 known_hosts：`ssh-keyscan -H <host>`
```
